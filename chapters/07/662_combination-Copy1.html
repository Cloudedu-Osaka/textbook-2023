def decimal_to_sexagesimal(decimal):
    integer_part = int(decimal)
    fractional_part = decimal - integer_part
    sexagesimal_integer = []
    while integer_part > 0:
        remainder = integer_part % 60
        sexagesimal_integer.append(remainder)
        integer_part //= 60
    sexagesimal_integer.reverse()
    sexagesimal_fractional = []
    while fractional_part > 0 and len(sexagesimal_fractional) < 10:
        product = fractional_part * 60
        integer_part = int(product)
        sexagesimal_fractional.append(integer_part)
        fractional_part = product - integer_part
    return sexagesimal_integer, sexagesimal_fractional
