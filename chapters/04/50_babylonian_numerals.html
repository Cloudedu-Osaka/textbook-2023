def decimal_to_babylonian(num):
    result = ''
    for value in sorted(babylonian_numerals.keys(), reverse=True):
        while num >= value:
            result += babylonian_numerals[value]
            num -= value
    return result

def decimal_to_sexagesimal(decimal):
    sexagesimal = []
    integer_part = int(decimal)
    decimal_part = decimal - integer_part

    while integer_part > 0:
        remainder = integer_part % 60
        sexagesimal.insert(0, remainder)
        integer_part //= 60

        decimal_part *= 60
        sexagesimal.append(int(decimal_part))
        decimal_part -= int(decimal_part)

    return sexagesimal
